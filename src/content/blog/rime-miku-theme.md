快来使用Rime初音未来主题！

```yaml title='weasel.yaml'
miku:
  name: 初音未来
  author: NorthZeroD
  # 背景设置
  back_color: 0xAFBBC539                        # 窗口背景
  border_color: 0xFFBBC539                      # 窗口边框
  shadow_color: 0x10BBC539                      # 窗口阴影
  # 内选区域
  text_color: 0xFFA9DEFF                        # 文字
  hilited_text_color: 0xFFA9DEFF                # 编码
  hilited_back_color: 0x00000000                # 背景
  hilited_shadow_color: 0x00000000              # 阴影
  # 以下两项在 inline_preedit: false 时生效
  nextpage_color: 0xFF6300D1                    # 翻页箭头颜色：下一页；不设置则不显示箭头
  prevpage_color: 0xFF6300D1                    # 翻页箭头颜色：上一页；不设置则不显示箭头
  # 激活候选项
  hilited_mark_color: 0x00000000                # 激活标签
  hilited_label_color: 0xFF6300D1               # 序号文字
  hilited_candidate_text_color: 0xFF6300D1      # 候选文字
  hilited_comment_text_color: 0x00000000        # 提示文字
  hilited_candidate_back_color: 0xAFFFFFFF      # 激活背景
  hilited_candidate_border_color: 0xFFBBC539    # 激活边框
  hilited_candidate_shadow_color: 0x00000000    # 激活阴影
  # 其它候选项
  label_color: 0xFFFFFFFF                       # 序号文字
  candidate_text_color: 0xFFFFFFFF              # 候选文字
  comment_text_color: 0x00000000                # 提示文字
  candidate_back_color: 0x00000000              # 候选背景
  candidate_border_color: 0x00000000            # 候选边框
  candidate_shadow_color: 0x00000000            # 候选阴影
```
